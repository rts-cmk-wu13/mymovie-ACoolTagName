const options = {
	threshold: 1.0;
}

const observer = new IntersectionObserver(function(entries) {
	entries.forEach(function(entry) {
	if(entry.isIntersecting) {
		entry.target.classList.add("theOne--visible");
		observer.unobserve(entry.target)
	};
	})


}, options)

let theOneElm = document.querySelector(".theone")

observer.observe(theOneElm)
